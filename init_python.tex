\begin{frame}[fragile]
\frametitle{Une définition de la Programmation}
\begin{itemize}[<+->]
  \item En partant d'un \textbf{algorithme} qui réalise une tâche donnée, on écrit un \textbf{logiciel} ou un \textbf{programme} à l'aide d'un langage informatique, qui lorsque qu'on l'exécute sur un ordinateur, réalise effectivement cette tâche.
  \item le programme prend souvent des données en entrée, effectue un traitement, et retourne des données en sortie. Le 
  traitement en question des données est la \textbf{valeur ajoutée} du programme.
 \item on peut schématiser cela grossièrement par :

\begin{frame}[fragile]
\frametitle{Exemple : somme des $n$ premiers entiers}
\begin{itemize}[<+->]
  \item l'algorithme peut ressembler à ça : 
   \begin{description}
      \item[1.] dans une case mémoire (variable), je mets 1
      \item[2.] au contenu de la case mémoire, j'ajoute 2
      \item[3.] au contenu de la case, j'ajoute 3 
      \item[...] .....
      \item[n.] au contenu de la case mémoire, j'ajoute $n$ 
      \item[n+1.] je retourne le contenu de la case mémoire
   \end{description}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Exemple : somme des $n$ premiers entiers}
\begin{itemize}
  \item On peut formaliser l'{\bf algorithme} avec du {\em pseudo-code}
   \begin{algorithmic}
   \State{$s \gets 1$}
   \For{$i= 2,\cdots,n$} 
     \State{$s \gets s + i$}
   \EndFor
   \State{afficher $s$}
   \end{algorithmic}
\item Ce qui donne en \python
\begin{lstlisting} 
    s = 1  
    for i in range(n):
        s = s + i
    print s
\end{lstlisting}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Structures de données}
\begin{itemize}[<+->]
   \item pour manipuler des données, on a besoin de \textbf{structures de données}
   \item nombres : 
       \begin{itemize}
         \item entiers : \texttt{n = 2} 
         \item flottants : \texttt{z = 1.3225, b=1/3.}
       \end{itemize}
   \item textes : chaînes de caractères 
       \begin{itemize}
        \item par exemple, \texttt{ a = '123'} ou \texttt{ s = "coucou"}
       \end{itemize}
   \item listes : 
       \begin{itemize}
          \item \texttt{l  =  [1, '2', -5] ; print l[2]} 
          \item \texttt{v = range(1,10) \# v = [1, ..., 9]}
       \end{itemize}
   \item dictionnaires : 
     \begin{itemize}
      \item \texttt{dic= \{ 'a':'coucou', 12:[1,2,3] \} }
      \item dic['a'] contient 'coucou' et \texttt{dic[12][0]} contient 1
      \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Structure de contrôle : branchement}
\begin{itemize}
\item On peut avoir besoin dans un programme, selon un résultat
d'opter entre 2 ou plusieurs choix. 
\begin{columns}
\column{0.5\linewidth}
\begin{algorithmic}
\If{ $x < 1.5$} 
    \State{afficher 'je suis petit'} 
\ElsIf{$ x < 1.8$}
    \State{afficher 'je suis moyen'} 
\Else
    \State{afficher 'je suis grand'} 
\EndIf
\end{algorithmic}
\column{0.5\linewidth}
\begin{lstlisting}
if x < 1.50 :
   print('je suis petit')
elif  x < 1.80 :
   print('je suis moyen')
else:
   print('je suis grand')
\end{lstlisting}
\end{columns}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Structure de contrôle : les fonctions}
\begin{itemize}
\item Permet de découper un programme en sous tâches. 
\item prend des arguments (liste d'arguments en parenthèses)
\item renvoie un résultat (mot-clef \texttt{return})
\begin{columns}
\column{0.5\linewidth}
\begin{algorithmic}
\Function{est\_pair}{$x$}
   \If {$x \bmod 2  = 0$}
       \State{{\ECFAugie retourner} vrai}
    \Else 
       \State{{\ECFAugie retourner} faux}
   \EndIf
\EndFunction
\end{algorithmic}
\column{0.5\linewidth}
\begin{lstlisting} 
def est_pair(x):
    if x % 2 == 0 : 
        return True
    else :
        return False
\end{lstlisting} 
\end{columns}
\item on peut aussi déclarer une fonction {\em anonyme} avec \texttt{lambda} : 
\begin{lstlisting} 
(lambda x: x*x)(2) # affiche 4
\end{lstlisting} 
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Code César et messages secrets}
\begin{itemize} 
  \item on veut coder un message en utilisant le code de César : on décale chaque lettre de l'alphabet d'un nombre donné
  à l'avance
  \item exemple : si la clef est 3, a devient d , z devient c \\
\begin{center} 
  \begin{tabular}{||c|c|c|c|c|c||}
 \hline 
 \hline
   a & b & c & $\cdots$ & y & z \\
   \hline
   d & e & f & $\cdots$ & b & c \\
  \hline 
  \hline
  \end{tabular}
  \end{center}
\item à l'aide d'un dictionnaire, on mesure la ``vraisemblance d'une clef'' pour un message comme la somme des mots du message qui décodés sont dans le dictionnaire
  \item on choisit alors la clef dont la vraisemblance est maximum
\end{itemize}
\end{frame}

\frametitle{Animation graphique}
\begin{itemize} 
   \item une sphère de centre $(x,y,z)$ rebondit à l'intérieur d'un parallélépipède $[0,L]\times [0,H] \times [0,P]$ de la façon suivante
   $$\begin{cases} 
   x \leftarrow x + v_x \times dt \\
   y \leftarrow y + v_y \times dt \\
   z \leftarrow z + v_z \times dt \\
   \end{cases}
   $$
  \item si on touche un bord, par exemple $y <0 $ alors $v_y \leftarrow -v_y, y \leftarrow 0 $ et on continue
\end{itemize}
