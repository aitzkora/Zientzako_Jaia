\begin{frame}
\frametitle{Exemple : somme des $n$ premiers entiers}
\begin{itemize}[<+->]
  \item l'algorithme peut ressembler a ca : 
   \begin{description}
      \item[1.] dans une case mémoire (variable) je mets 1
      \item[2.] au contenu de la case mémoire, j'ajoute 2
      \item[3.] au contenu de la case, j'ajoute 3 
      \item[...] .....
      \item[n.] au contenu de la case mémoire, j'ajoute $n$ 
      \item[n+1.] je retourne le contenu de la case mémoire
   \end{description}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Exemple : somme des $n$ premiers entiers}
\begin{itemize}
  \item On peut formaliser l'{\bf algorithme} avec du {\em pseudo-code}
   \begin{algorithmic}
   \State{$s \gets 1$}
   \For{$i= 2,\cdots,n$} 
     \State{$s \gets s + i$}
   \EndFor
   \State{afficher $s$}
   \end{algorithmic}
\item Ce qui donne en \python
\begin{lstlisting} 
    s = 1  
    for i in range(n):
        s = s + i
    print s
\end{lstlisting}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Caracteristiques du langage \python}
\begin{itemize}
  \item \python est un langage interprété : on peut modifier
  dynamiquement (à la volée) le code du programme.
  \item la structure du programme se gère grâce à des indentations : attention à aligner
  les blocs avec le même nombre d'espace
\vspace{1cm}
\begin{columns}
\column{0.4\linewidth}
{\bf \textcolor{blue}{Bien}} 
\begin{lstlisting}
i = 1  
while i < 10 : 
    i = i * 2
    j = i - 2
\end{lstlisting}
\column{0.4\linewidth}
{\bf \textcolor{red}{Pas bien}}
\begin{lstlisting}
  i = 1  
while i < 10 : 
    i = i * 2
  j = i - 2
\end{lstlisting}
\end{columns}
\end{itemize}
\end{frame}

\begin{frame}[fragile]

\begin{itemize}
\item On peut avoir besoin dans un programme, selon un résultat
d'opter entre 2 ou plusieurs choix.

\If{ $x < 1.5$} 
    \State{afficher 'je suis petit'} 
\ElsIf{$ x < 1.8$}
    \State{afficher 'je suis moyen'} 
\Else
    \State{afficher 'je suis grand'} 
\EndIf
\end{algorithmic}
\column{0.5\linewidth}
\begin{lstlisting}
if x < 1.50 :
   print('je suis petit')
elif  x < 1.80 :
   print('je suis moyen')
else:
   print('je suis grand')
\end{lstlisting}
\end{columns}
\end{itemize}

\begin{frame}[fragile]
\frametitle{Structure de contrôle : les fonctions}
\begin{itemize}
\item Permet de découper un programme en sous tâches. 
\item prend des arguments (liste d'arguments en parenthèses)
\item renvoie un résultat (mot-clef return)
\begin{columns}
\column{0.5\linewidth}
\begin{algorithmic}
\Function{est\_un\_carre}{$x$}
   %\If {$\sqrt{x} \dot \sqrt{x} = x$}
   \If { $\sqrt{x} \times \sqrt{x}= x$}
       \State{\textbf{retourner} vrai}
    \Else 
       \State{\textbf{retourner} faux}
   \EndIf
\EndFunction
\end{algorithmic}
\column{0.5\linewidth}
\begin{lstlisting} 
def est_un_carre(x):
    if sqrt(x)*sqrt(x) == x : 
        return True
    else :
        return False
\end{lstlisting} 
\end{columns}
\end{itemize}
\end{frame}
